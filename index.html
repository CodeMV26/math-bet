    <script>

        const odds = 1.65;

        const probability = 0.635; // 63.5%

        

        // Criterio di Kelly frazionario (usiamo Kelly 1/1 per scopi didattici)

        // Formula: f* = (p * q - 1) / (q - 1)

        const kellyFraction = ((probability * odds) - 1) / (odds - 1); // 0.07346 (7.346%)


        const bankrollInput = document.getElementById('bankroll');

        const kellyPercentSpan = document.getElementById('kelly-percentage');

        const kellyStakeSpan = document.getElementById('kelly-stake');

        

        const resultBox = document.getElementById('result-box');

        const grossReturnSpan = document.getElementById('gross-return');

        const netProfitSpan = document.getElementById('net-profit');

        const newBankrollSpan = document.getElementById('new-bankroll');


        function calcolaPuntata() {

            let bankroll = parseFloat(bankrollInput.value) || 0;

            if (bankroll < 0) bankroll = 0;

            

            const stake = bankroll * kellyFraction;

            

            kellyPercentSpan.innerText = (kellyFraction * 100).toFixed(2) + "%";

            kellyStakeSpan.innerText = stake.toFixed(2) + " €";

        }


        function simulaRisultato() {

            let bankroll = parseFloat(bankrollInput.value) || 0;

            if (bankroll <= 0) return;


            const stake = bankroll * kellyFraction;

            const profit = stake * (odds - 1);

            const totalReturn = stake * odds;

            const finalBankroll = bankroll + profit;


            grossReturnSpan.innerText = totalReturn.toFixed(2) + " €";

            netProfitSpan.innerText = "+" + profit.toFixed(2) + " €";

            newBankrollSpan.innerText = finalBankroll.toFixed(2) + " €";


            resultBox.classList.remove('hidden');

            resultBox.scrollIntoView({ behavior: 'smooth' });

        }


        // Ascolta l'inserimento dell'utente per aggiornare la scommessa in tempo reale

        bankrollInput.addEventListener('input', calcolaPuntata);

        

        // Esegui calcolo iniziale

        calcolaPuntata();

    </script>
